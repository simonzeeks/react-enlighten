## Install

```bash
npm install --save react-enlighten
```

## Usage

```jsx
import React, { Component } from "react";

import EnlightenedComponent from "react-enlighten";

class Example extends EnlightenedComponent {
  render() {
    return <MyComponent />;
  }
}
```
